\chapter{高階導関数}

\section{高階導関数}

\textbf{1. 高階導関数}

関数$y=f(x)$の導関数$f'(x)$が微分可能であるとき，
$f'(x)$の導関数を$f''(x)$で表して，$f(x)$の\textbf{2階導関数}という;すなわち，
\[
	f''(x)	=	\{f'(x)\}'.
\]
2階導関数を表す記号としては，$f''(x)$のほかに，
\[
	y''	,	\quad	\frac{d^2y}{dx^2},			\quad		\{f(x)\}'',
				\quad	\frac{d^2	f(x)}{dx^2},	\quad		\frac{d^2}{dx^2}f(x),
				\quad	D^2	f(x)
\]
などが用いられる．

帰納的に，$(n-1)$階導関数：
$f^{(n-1)}(x)	\quad	(n=1,2,\cdots,	;	f^{(0)}(x)=f(x))$が微分可能であるとき；
すなわち，$f(x)$が\textbf{$n$階微分可能}であるとき，
$f^{(n-1)}(x)$の導関数を$f^{(n)}$で表して，$f(x)$の\textbf{$n$階}(または
\textbf{$n$次)導関数}という；すなわち，
\[
	f^{(n)}(x)=\{f^{(n-1}(x)\}'	\quad	(n=1,2,\cdot).
\]
$n$階導関数を表す記号としては，$f^{(n)}(x)$のほかに，
\[
	y^{n}	,	\quad	\frac{d^ny}{dx^n},		\quad		\{f(x)\}^{(n)},
				\quad	\frac{d^n	f(x)}{dx^n},	\quad		\frac{d^n}{dx^n}f(x),
				\quad	D^n	f(x)
\]
などが用いられる．一般に，2階以上の導関数を総称して，\textbf{高階}(または\textbf{高次})
\textbf{導関数}という．

\textbf{3. 2項定理}

\begin{equation}
	(1+x)^n	=	a_0	+	a_1	x	+	\cdots	+	a_n	x^n
	\label{eq11.2}
\end{equation}
の係数：$a_0,\cdots	,	a_n$を決定しよう．

まず，(\ref{eq11.2})式の両辺において，$x=0$とおけば，
\begin{equation}
	a_0=1.
	\label{eq11.3}
\end{equation}
$f(x)=(1+x)^n$とおけば，公式(\ref{eq11.1})によって，
\[
	f^{(k)}(x)	=	n(n-1)\cdots	(n-k+1)(1+x)^{n-k}	\quad	(k=1,\cdots,	n).
\]
\begin{equation}
	\therefore	f^{(k)}(0)	=	n(n-1)\cdots	(n-k+1)	\quad	(k=1,\cdots,	n).
	\label{eq11.4}
\end{equation}
一方，(\ref{eq11.2})式の右辺を$k$回微分すれば，公式(\ref{eq11.1})によって，
\[
	f^{(k)}=k!	a_k	(k+1)	k	\cdots	2	a_{k+1}	x	+	\cdots
																							+	n(n-1)	\cdots	(n-k+1)a_nx^{n-k}.
\]
\begin{equation}
	\therefore	f^{(k)}(0)	=	k!	a_k	\quad	(k=1,\cdots,	n).
	\label{eq11.5}
\end{equation}
(\ref{eq11.4})式と(\ref{eq11.5})式によって，
\[
	a_k	=	\frac{n(n-1)	\cdots	(n-k+1)}{k!}	=	\binom{n}{k}	=	{}_{n}\text	C_{k}
	\quad	(k=1,\cdots	,	n).
\]
したがって，
\[
	\binom{n}{0}	\equiv	1
\]
と約束すれば，
\begin{equation}
	(1+x)^n	=	\binom{n}{0}		+	\binom{n}{1}	x	+	\cdots	+
						\binom{n}{k}x^k	+	\cdots	+	\binom{n}{n}	x^n
	\label{eq11.6}
\end{equation}
この式に$\displaystyle	x	=	\frac{b}{a}$を代入し，両辺に$a^n$をかければ，
\begin{equation}
	(a+b)^n	=	\binom{n}{0}a^n		+	\binom{n}{1}	a^{n-1}b	+	\cdots	+
						\binom{n}{k}a^{n-k}b^k	+	\cdots	+	\binom{n}{n}	b^n.
		\quad	\text{\textbf{(2項定理)}}
		\label{eq11.7}
\end{equation}

	\label{eq11.9}

\[
	(\sin	x)'	=	\cos	x	=	\sin	\left(x	+	\frac{\pi}{2}	\right),
\]
となり成立する．$n=k$のとき成立すると仮定すると
\[	
	(\sin	x)^{(k)}	=	\sin	\left(x	+	\frac{k\pi}{2}	\right).
\]
$n=k+1$のとき
\[	
	(\sin	x)^{(k+1)}	=	\left(\sin	\left(x	+	\frac{k\pi}{2}	\right)\right)'
										=	\cos	\left(x	+	\frac{k\pi}{2}	\right)
										=	\sin	\left(x	+	\frac{k\pi}{2}	+\frac{\pi}{2}	\right)
										=	\sin	\left(x	+	\frac{(k+1)\pi}{2}	\right).
\]
以上より，$\displaystyle	(\sin	x)^{(n)}	=	\sin	\left(	x	+	\frac{n\pi}{2}	\right)$.

\問{6}
$y=f(u)$，$u=g(x)$が，それぞれ2階導関数をもつとき，合成関数$y=f(g(x))$に対して，
つぎの公式が成り立つkとを証明せよ．
\[
	\frac{d^2	y}{dx^2}	=		\frac{d^2	y}{du^2}	\left(\frac{du}{dx}\right)^2
												+	\frac{dy}{du}\cdot	\frac{d^2	u}{dx^2}.
												\quad	\quad	\text{(\textbf{合成関数の2次微分の公式})}
\]
\証明
$\displaystyle	\frac{dy}{dx}=\frac{dy}{du}\cdot	\frac{du}{dx}$であるから，
\begin{align*}
	\frac{d^2y}{dx^2}	&=	\frac{d}{dx}	\left(\frac{dy}{du}	\right)\cdot	\frac{du}{dx}
										+	\frac{dy}{du}	\cdot\frac{d^2u}{dx^2}.	\quad\text{(積の微分)}	\\
										&=	\frac{d}{du}	\left(\frac{dy}{du}	\right)\cdot	\frac{du}{dx}\cdot\frac{du}{dx}
										+	\frac{dy}{du}\cdot\frac{d^2u}{dx^2}	\\
										&=	\frac{d^2y}{du^2}	\cdot	\left(\frac{du}{dx}	\right)^2
										+	\frac{dy}{du}	\cdot	\frac{d^2u}{dx^2}.
\end{align*}
(証明終)

\問{7}
単調で2回微分可能な関数$x	=	f(y)$の逆関数$y=f^{-1}(x)$に対して，
次の公式が成り立つことを証明せよ：
\[
	\frac{d^2y}{dx^2}	=	-\frac{	\displaystyle	\frac{d^2x}{dy^2}}{
															\displaystyle	\left(	\frac{dx}{dy}	\right)^3}.
											\quad	\text{(\textbf{逆関数の２次微分の公式})}
\]
\証明
$\displaystyle	\frac{dy}{dx}	=	\frac{1}{\displaystyle	\frac{dx}{dy}}$であるから，
\begin{align*}
	\frac{d^2y}{dx^2}	&	=	\frac{d}{dx}	\frac{dy}{dx}
											=	\frac{d}{dx}	\frac{1}{\displaystyle	\frac{dx}{dy}}
											=	\frac{d}{dy}	\frac{1}{\displaystyle	\frac{dx}{dy}}
																			\cdot	\frac{dy}{dx}
											=	-	\frac{1}{\displaystyle	\left(\frac{dy}{dx}\right)^2}
													\cdot	\frac{d^2x}{dy^2}	\cdot	\frac{dy}{dx}
											=	-	\frac{\displaystyle	\frac{d^2x}{dy^2}}{
																\displaystyle	\left(\frac{dy}{dx}\right)^3}
\end{align*}
(証明終)

\問{8}
$x=x(t)$，$y=y(t)$が2回微分可能であって，$\dot{x}(t)	\ne	0$であるとき
つぎの公式がなりたつことを証明せよ：
\[
	\frac{d^2y}{dx^2}	=	\frac{\dot{x}\ddot{y}	-\dot{y}\ddot{x}}{\dot{x}^3},
	\quad	\text{\textbf{(パラメター表示された関数の2次微分の公式)}}
\]
\証明
$\displaystyle	\dot{x}	=\frac{dx}{dt},\quad	\ddot{x}	=	\frac{d^2x}{dt^2},
	\quad	\frac{dy}{dx}=\frac{\dot{y}}{\dot{x}}$であるから，
\begin{align*}
	\frac{d^2y}{dx^2}	&	=	\frac{d}{dx}	\frac{dy}{dx}
											=	\frac{d}{dt}	\frac{dy}{dx}	\cdot	\frac{dt}{dx}
											=	\frac{d}{dt}	\frac{\dot{y}}{\dot{x}}	\cdot	\frac{dt}{dx}
											=	\frac{\ddot{y}\dot{x}	-	\dot{y}\ddot{x}}{(\dot{x})^2}
												\cdot	\frac{1}{\displaystyle	\frac{dx}{dt}}
											=	\frac{\ddot{y}\dot{x}	-	\dot{y}\ddot{x}}{(\dot{x})^3}
\end{align*}
(証明終)
