\textbf{1.　Rolleの定理}

\定理{13.1}
$f(x)$は$[a,b]$で連続，$(a,b)$で微分可能で，
\begin{equation}
	f(a)	=	f(b)	=	0
\end{equation}
ならば
\begin{equation}
	f'(c)=0	\quad	(a<c<b)
\end{equation}

\begin{align}
	f(c+h)	-	f(c)	\leqq	0	\quad	(h>0;	\text{図\ref{fig13.2}})	\notag	\\
	\therefore	\quad	
	f_+'(c)	\equiv	\lim_{h\rightarrow	+0}	\frac{f(c+h)-f(c)}{h}	\leqq	0.
	\label{eq13.3}
\end{align}
また，
\begin{align}
	f(c+h)	-	f(c)	\leqq	0	\quad	(h<0;\text{図\ref{fig13.2}})
	\notag	\\
	\therefore	\quad	
	f_-'(c)	\equiv	\lim_{h\rightarrow	+0}	\frac{f(c+h)-f(c)}{h}	\geqq	0.
	\label{eq13.4}
\end{align}
$f(x)$は$x=c$で微分可能であるから，
\begin{equation}
	\therefore	\quad	
		f'(c)=f_+'(c)	=	f_-'(c).
		\label{eq13.5}
\end{equation}
(\ref{eq13.3}),(\ref{eq13.4}),(\ref{eq13.5})式によって
\[
	f'(c)=0
\]
$f(x)$の値が負になるところがあれば，$f(x)$が最小となる$c	\quad	(a	<	c	<	b)$が存在し，
\begin{align*}
	f(c+h)-f(c)	\geqq	0	\quad	(h	>	0),	\quad	f(c+h)-f(c)	\geqq	0	\quad	(h	<	0)	
\end{align*}
したがって
\begin{align*}
	f_+'(c)	=	\lim_{h\rightarrow	+0}\frac{f(c+h)-f(c)}{h}	\geqq	0	\quad	(h	>	0),
	\quad
	f_-'(c)	=	\lim_{h\rightarrow	-0}\frac{f(c+h)-f(c)}{h}	\leqq	0	\quad	(h	<	0)	
\end{align*}
よって，$f'(c)=0$が示される(図\ref{fig13.3})
