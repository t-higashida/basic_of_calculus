\section{連続関数}

\subsection{連続関数}

関数$f(x)$が，つぎの３つの条件を満たすとき，
$f(x)$はつぎの条件をみたすとき，$f(x)$は
\textbf{点$x=0$で連続}であるという：
\begin{list}{}{}
	\item	（ⅰ）$x=a$は$f(x)$の\textbf{定義域}（定義されている点の集合）に属している
							($f(a)$が存在する）；
	\item	（ⅱ）$\displaystyle	\lim_{x\rightarrow	a}f(x)$が存在する；
	\item	（ⅲ）$\displaystyle	\lim_{x\rightarrow	a}f(x)$と$f(a)$が等しい；$\displaystyle	\lim_{x\rightarrow	a}f(x)=f(a)$（図\ref{fig 1.4}）．
\end{list}

関数$f(x)$が\textbf{開区間}$(a,b)$ ($a<x<b)$をみたす$x$の範囲)のすべての点で
連続であるとき，$f(x)$は\textbf{開区間$(a,b)$で連続}であるという．

関数$f(x)$が\textbf{閉区間}$[a,b]$ ($a\leqq	x	\leqq	b)$をみたす$x$の範囲)で定義された
関数$f(x)$が，$(a,b)$で連続であって，
\[
	f(a+0)=f(a),	\quad	f(b-0)=f(b)
\]
であるとき，$f(x)$は\textbf{閉区間$[a,b]$で連続}であるという．

\定理{1.2.1}
$f(x),g(x)$が$x=a$で連続ならば，
\begin{list}{}{}
	\item	（ⅰ）$c(f)	\quad$	(\text{$c$は定数});	
	\item	（ⅱ）$f(x)+g(x)	\quad	f(x)-g(x);$
	\item	（ⅲ）$f(x)g(x);$
	\item	（ⅳ）$\frac{f(x)}{g(x)}	\quad	(g(a)	\ne	0)$
\end{list}
は，それぞれ，$x=a$で連続である．\textbf{(連続関数の基本定理)}

\証明
定理1.1.1で，$\alpha=f(a),\:	\beta=g(a)$とおけばよい．
\■

\subsection{合成関数}

関数$u=f(x)$の\textbf{値域}(関数の値の集合)を含む範囲で
定義された関数$y=g(u)$に対して，\textbf{合成関数}
$\:	y=g	\circ	f(x)=g(f(x))$が定義される．

\定理{1.2.2}
$u=f(x)$が$x=a$で連続で，$g(u)$が$b=f(a)$で連続ならば，
合成関数$g(f(x))$は$x=a$で連続である；すなわち，
\[
	\lim_{x	\rightarrow	a}	g(f(x))	=	g(f(a)).
\]
\証明
$u=f(x)$は$x=a$で連続だから，
\[
	\lim_{x	\rightarrow	a}	f(x)	=	f(a)=b.
\]
$y=g(u)$は$u=b$で連続だから，
\[
	\lim_{u	\rightarrow	b}	g(u)	=	g(b)
\]
しがたって，
\[
	\lim_{x	\rightarrow	a}	g(f(x))	=	g(b)	=	g(f(a)).
\]
\■
