\section{平均値の定理}

\textbf{1.　平均値の定理}

\定理{14.1}\textbf{(平均値の定理)　}
$f(x)$が$[a,b]$で連続で，$(a,b)$で微分可能ならば，
\begin{equation}
	\frac{f(b)-f(a)}{b-a}	=	f'(c)	\quad	(a<c<b)
	\label{eq14.1}
\end{equation}
をみたす$c$が存在する(図\ref{fig14.1})

とおくとき，$F(x)$はRolleの定理の仮定“$F(a)=F(b)=0$，$F(x)$は$[a,b]$で連続，$(a,b)$で微分可能”をみたす(図\ref{fig14.2}).
したがって，Rolleの定理によって，
\begin{equation}
	F'(c)	=	0	\quad	(a	<	c	<	b)
	\label{eq14.3}
\end{equation}
をみたす$c$が存在する．(\ref{eq14.2})によって，
\begin{equation}
	F'(c)	=	f'(c)	-	m	=	0
	\label{eq14.4}
\end{equation}
であるから，(\ref{eq14.3})式と(\ref{eq14.4})式によって，
\[
	0	=	F'(c)	=	f'(c)	-m	=	f'(c)	-\frac{f(b)-f(a)}{b-a}
	\quad	(a	<	c	<	b)
\]

(証明終)

\textbf{2.　平均値の定理の変形}

さて，(\ref{eq14.1})式は，$a$と$b$の役割を取り替えることによって，
$a	>	b$のときも，
\begin{equation}
	\frac{f(b)-f(a)}{b-a}=	f'(c)	\quad	(a	>	c	>	b)
	\label{eq14.5}
\end{equation}
の形で成り立つことがわかる．

したがって，平均値の定理は，
\begin{equation}
	f(b)=f(a)+(b-a)	f'(c)	\quad	(a	\gtrless	c	\gtrless		b)	\quad\text{\textbf{平均値の定理}}
	\label{eq14.6}
\end{equation}
の形に書くことができる．ここで，
\[
	\theta	=	\frac{c-a}{b-a}
\]
とおけば，$a$と$b$の大小関係に係わらず，
\[
	0	<	\theta	<	1
\]
であって，(\ref{eq14.6})式は，
\begin{equation}
	f(b)=f(a)	+	(b-a)	f'(a+\theta	(	b-a))	\quad	(0	<	\theta	<	1)
\end{equation}
と書ける．ここで，さらに，$b=a+h$とおけば，
\begin{equation}
	f(a+h)=f(a)	+	f'(a	+	\theta	h)	h	\quad	(0	<	\theta	<	1)
	\quad\quad	\text{\textbf{(平均値の定理)}}
\end{equation}
と書ける；ここで，$h$は正，負，いずれでもよいことに注意．

\textbf{3.　平均値の定理の応用}

\定理{14.2}$f(x)$が$[a,b]$で連続で，$(a,b)$で
\[
	f'(x)	\equiv	0
\]
ならば，
\[
	f(x)	\equiv	c	\quad	\text{：$c$は定数}
\]
である．

\証明
平均値の定理によって，任意の$x	\in	[a,b]$に対して，
\[
	f(x)	=	f(a)	+	(x-a)	f'(\xi)	\quad	(a	<	\xi	<	x)
\]
をみたす$\xi$が存在する．定理の仮定によって，$f'(\xi)=0$であるから，
\[
	f(x)	=	f(a).
\]
$x$は任意であったから，
\[
	f(x)	\equiv	f(a).
\]
(証明終)

\textbf{4．コーシーの平均値の定理}

\定理{14.3}
$f(x)$，$g(x)$が$[a,b]$で連続，$(a,b)$で微分可能で，$g'(x)\ne	0$ならば，
\begin{equation}
	\frac{f(b)-f(a)}{g(b)-g(a)}	=	\frac{f'(c)}{g'(c)}	\quad	(a	<	c	<	b)
	\label{eq14.9}
\end{equation}
をみたす$c$が存在する．\textbf{(コーシーの平均値の定理)}

\証明
\begin{equation}
	F(x)	\equiv	\{g(b)-g(a)\}\{f(x)-f(a)\}	-	\{f(b)-f(a)\}\{g(x)-g(a)\}
	\label{eq14.10}
\end{equation}
とおけば，
\[
	F(a)	=	F(b)	=	0
\]
となり，$F(x)$は，Rolleの定理の仮定をみたす．
したがって，Rolleの定理によって，
\begin{equation}
	F'(c)	=	0	\quad	(a	<	c	<	b)
	\label{eq14.11}
\end{equation}
をみたす$c$が存在する．(\ref{eq14.10})式によって，
\begin{equation}
	F'(x)	=	\{g(b)	-	g(a)	\}	f'(x)	-	\{f(b)	-	f(a)	\}g'(x)
	\label{eq14.12}
\end{equation}
であるから，(\ref{eq14.11})式と(\ref{eq14.12})式によって，
\begin{equation}
	0	=	F'(c)	=	\{g(b)	-	g(a)	\}	f'(c)	-	\{f(b)	-	f(a)	\}	g'(c)
	\quad
	(a	<	c	<	b).
	\label{eq14.13}
\end{equation}
定理の仮定：$g(x)	\ne	\:	(a	<	x	<	b)$から，平均値の定理によって，
$g(b)	-	g(a)	\ne	0$．そのとき，(\ref{eq14.13})式から(\ref{eq14.9})式が得られる．

(証明終)
